Mail:
	[:content:]

content:
    [:headers:][:body:]

headers:
    [:header:]*\n+

body:
    ([:b_start:][:content:])+[:b_end:]
    [:plain:]

header:
    [:key:]: [:values:]\n

b_start:
    --[:boundary:]\n

b_end:
    --[:boundary:]--\n

plain:
    A string that doesn't include [:boundary:].  [:plain:] must end with "\n".

key:
    [a-zA-Z-]+

values:
    ([:value:](;[:value:])*)?

value:
    [^;]+

boundary:
    A unique string that is predefined for each level.
